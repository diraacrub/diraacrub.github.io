    <!-- Lógica de JavaScript para la presentación -->
    <script>
        let currentSlide = 0;
        const slides = document.querySelectorAll('.slide');
        const totalSlides = slides.length;
        
        const counterElement = document.getElementById('slide-counter');
        const prevBtn = document.getElementById('prevBtn');
        const nextBtn = document.getElementById('nextBtn');

        function showSlide(n) {
            // Actualizar el índice de la diapositiva actual
            currentSlide = n;

            // Ocultar todas las diapositivas
            slides.forEach((slide, index) => {
                slide.classList.toggle('hidden', index !== currentSlide);
            });

            // Actualizar el contador
            counterElement.textContent = `${currentSlide + 1} / ${totalSlides}`;

            // Habilitar/deshabilitar botones de navegación
            prevBtn.disabled = (currentSlide === 0);
            nextBtn.disabled = (currentSlide === totalSlides - 1);
        }

        // --- Event Listeners ---

        // Botones de Click
        prevBtn.addEventListener('click', () => {
            if (currentSlide > 0) {
                showSlide(currentSlide - 1);
            }
        });

        nextBtn.addEventListener('click', () => {
            if (currentSlide < totalSlides - 1) {
                showSlide(currentSlide + 1);
            }
        });

        // Navegación con teclado (flechas)
        document.addEventListener('keydown', (e) => {
            if (e.key === 'ArrowRight') {
                if (currentSlide < totalSlides - 1) {
                    showSlide(currentSlide + 1);
                }
            } else if (e.key === 'ArrowLeft') {
                if (currentSlide > 0) {
                    showSlide(currentSlide - 1);
                }
            }
        });

        // Mostrar la primera diapositiva al cargar
        showSlide(0);
    </script>
